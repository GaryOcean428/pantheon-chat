private pearsonCorrelation(x: number[], y: number[]): number {
  const n = x.length;
  const sum_x = x.reduce((a, b) => a + b, 0);
  const sum_y = y.reduce((a, b) => a + b, 0);
  const sum_xy = x.reduce((sum, xi, i) => sum + xi * y[i], 0);
  const sum_x2 = x.reduce((sum, xi) => sum + xi * xi, 0);
  const sum_y2 = y.reduce((sum, yi) => sum + yi * yi, 0);
  
  const numerator = n * sum_xy - sum_x * sum_y;
  const denominator = Math.sqrt((n * sum_x2 - sum_x * sum_x) * (n * sum_y2 - sum_y * sum_y));
  
  return denominator === 0 ? 0 : numerator / denominator;
}

  resonantCandidates: number;
  discoveryRate4D: number;     // near-misses per hour in 4D
  discoveryRate3D: number;     // near-misses per hour in 3D
  
  // Validation
  hypothesis: string;          // e.g., "4D improves discovery by 10x"
  result: 'confirmed' | 'rejected' | 'inconclusive';
  evidence: string;
}

// Compare 3D vs 4D performance
export function comparePerformance(run: ExperimentRun): {
  advantage4D: number;         // Factor improvement
  statistical_significance: number;
} {
  const rate3D = run.discoveryRate3D;
  const rate4D = run.discoveryRate4D;
  
  if (rate3D === 0 || rate4D === 0) {
    return { advantage4D: 0, statistical_significance: 0 };
  }
  
  const advantage = rate4D / rate3D;
  
  // Simple significance test (would need more data for real p-value)
  const significance = Math.min(1, advantage / 2);
  
  return { advantage4D: advantage, statistical_significance: significance };
}
